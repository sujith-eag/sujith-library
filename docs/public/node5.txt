const fs = require('fs');

const readStream = fs.createReadStream('data.txt', 'utf8');

readStream.on('data', (chunk) => {
  console.log('--- New Chunk Received ---');
  console.log(chunk);
});

readStream.on('end', () => {
  console.log('\n--- Finished Reading File ---');
});

readStream.on('error', (err) => {
    console.error('An error occurred:', err);
});
