const fs = require('fs');

function add(a, b, callback) {
  const sum = a + b;
  callback(sum);
}

add(10, 20, (val) => {
  const result = val * 10;
  console.log(`Result after multiplication: ${result}`);
});

fs.readFile('info.txt', 'utf8', (err, data) => {
  if (err) {
    console.error("Error reading file:", err);
    return;
  }
  console.log("\nContents of info.txt:");
  console.log(data);
});
